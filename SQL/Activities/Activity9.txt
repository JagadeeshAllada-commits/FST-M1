SELECT * FROM SALESMAN;

SELECT c.customer_id, C.customer_name, s.salesman_id, s.salesman_name 
    FROM Customers c Join Salesman s 
    on c.salesman_id = s.SALESMAN_ID;

SELECT c.customer_id, C.customer_name,c.GRADE, s.salesman_id, s.salesman_name 
    FROM Customers c Join Salesman s 
    on c.salesman_id = s.SALESMAN_ID
    WHERE c.grade < 300 ORDER BY c.CUSTOMER_ID asc;


SELECT c.customer_id, C.customer_name,s.commission , s.salesman_id, s.salesman_name 
    FROM Customers c Join Salesman s 
    on c.salesman_id = s.SALESMAN_ID
    WHERE s.commission > 12;

SELECT o.order_no, o.order_date, o.purchase_amount, 
    c.customer_id, c.customer_name, 
    s.salesman_id, s.salesman_name 
    from Orders o JOIN Customers c on o.CUSTOMER_ID = c.CUSTOMER_ID
    JOIN Salesman s on c.salesman_id = s.salesman_id;
